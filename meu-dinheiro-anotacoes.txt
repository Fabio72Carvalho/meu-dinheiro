import { salvarConta } from './db.js';

// Elementos da seção contas
const secaoContas = document.getElementById('secao-contas');
const btnSalvarConta = document.getElementById('btn-salvar-conta');
const inputNomeConta = document.getElementById('nome-conta');
const inputSaldoInicial = document.getElementById('saldo-inicial');


// --- EVENTOS DE BANCO DE DADOS (Firestore) ---
if (btnSalvarConta) {
    btnSalvarConta.addEventListener('click', async () => {
        const nome = inputNomeConta.value;
        const saldo = inputSaldoInicial.value;

            alert("Por favor, preencha todos os campos da conta.");

                alert("Conta '" + nome + "' cadastrada com sucesso!");

                inputNomeConta.value = '';
                inputSaldoInicial.value = '';

            } catch (error) {
                alert("Erro ao salvar conta: " + error.message);
            }
        } else {
            alert("Você precisa estar logado para cadastrar uma conta.");
        }
    });
