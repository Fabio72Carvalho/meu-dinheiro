if (btnSalvarConta) {
    btnSalvarConta.addEventListener('click', async () => {
        const nome = inputNomeConta.value;
        const saldo = inputSaldoInicial.value;

        if (!nome || !saldo) {
            alert("Por favor, preencha todos os campos da conta.");
            return;
        }

        if (usuarioAtual) {
            try {
                const dadosConta = {
                    nome: nome,
                    saldoAtual: parseFloat(saldo)
                };

                await salvarConta(usuarioAtual.uid, dadosConta);

                alert("Conta '" + nome + "' cadastrada com sucesso!");

                // Limpar campos
                inputNomeConta.value = '';
                inputSaldoInicial.value = '';

            } catch (error) {
                alert("Erro ao salvar conta: " + error.message);
            }
        } else {
            alert("Você precisa estar logado para cadastrar uma conta.");
        }
    });
}
